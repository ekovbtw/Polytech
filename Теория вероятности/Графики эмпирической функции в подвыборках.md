    `# 🔹 Запись в файл`
    `with open(filename, 'w') as f:`
        `f.write("x;F(x)\n")`

        `for i in range(n):`
            `x = X_sorted[i]`
            `F_prev = i / n`
            `F_curr = (i + 1) / n`

            `# 🔹 горизонтальный участок`
            `f.write(f"{x};{F_prev}\n")`

            `# 🔹 скачок вверх`
