short position = 0;  // указатель на место на экране
void str_str(char* point) // Функция вывода строки point - указатель на первый символ в строке
{
	volatile char* video = (volatile char*)0xB8000; // из методички (текст выводится через видеопамять с адреса 0xB8000)
	while (*point) // проход по строке
	{
		video[position*2] = *point; // запись символа в видеопамять (*2 тк сам символ + цвет)
		video[position*2 +1] = 0x07;
		position++; // сдвиг
		point++; // сдвиг
	}
}

void put_point (char point) // функция вывода одного символа (для алфавита)

	video[position*2+1] = 0x07;
	position++;
}
