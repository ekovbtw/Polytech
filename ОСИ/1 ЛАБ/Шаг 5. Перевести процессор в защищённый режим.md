В загрузчике: 
отключить прерывания;
загрузить GDT;
включить A20;
выставить бит PE в CR0; 
сделать дальний переход в 32-битный код

Потому что современное ядро должно работать не в реальном, а в защищённом режиме. В методичке прямо сказано, что задача загрузчика — перевести процессор в него.

`kernel_loaded:`
	`call turn_protect_mode`
	`jmp $` изменил эту функцию тк перед тем как запустить ядро нужно перевести процессор в защищенный режим

`turn_protect_mode:`
    	`cli` отключить прерывания
    	`lgdt [gdt_info]` загрузить глобальную таблицу дескрипторов [[О таблице дескрипторов]]
    	

	; A20 включение адресной линии(позволяет нормально адресовать память больше 1мб)
	in al, 0x92
	or al, 2
	out 0x92, al

	; CR0 включение PE protection enable (если равен 1 процессор переходит в защищенный режим)
	mov eax, cr0 нулевой бит CR0 отвечает за вход в защищенный режим
	or eax, 1
	mov cr0, eax

	jmp 0x08:protect_mode дальний прыжок тк мы почти в protect mode (так нужно)
вообще это все из методички

`gdt:`  
`db 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00`  
`db 0xff, 0xff, 0x00, 0x00, 0x00, 0x9A, 0xCF, 0x00`  
`db 0xff, 0xff, 0x00, 0x00, 0x00, 0x92, 0xCF, 0x00`  
`gdt_end:`  
  
`gdt_info:`  
`dw gdt_end - gdt - 1`  
`dd gdt`

`mov ds, ax`  
`mov es, ax`  
`mov fs, ax`  
`mov gs, ax`  
`mov ss, ax`  
  
`mov esp, 0x90000`
`mov eax, 0x10000`
`сall eax  вызов ядра теперь тут`
  
`hang:`  
`hlt`  
`jmp hang`
