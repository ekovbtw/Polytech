напишем функцию только для символа "a" :
`sub al, 'a'` 
	
	xor ah,ah ; ax = ah + al (al == 'index a ' - младший байт, ah - старший байт), поэтому сделаем обнуление старшего байта
	mov si,ax ;  записываем в si регистр значение ax = 0002 (00 = ah, 02 = al)
	
	mov bl, [turn_symbol_array+si] ; берем индекс буквы из массива
	xor bl, 1 ; меняем если 1 на 0, если 0 то 1
	mov [turn_symbol_array +si], bl ; заменяем
	
	mov al, 13 
	int 0x10
	mov al, 10 
	int 0x10 ; ENTER +space

теперь осталось обновить строку на экране: 
	`call update_selected_symbols` ; вызов обновление строки
	`jmp input_symbol_loop` ; после функции обновления цикл повторяется (ввод буквы)
	

`update_selected_symbols:`
	`mov ah, 0x0E`; обновляем регистр ah на прерывания чтобы выводить текст
	`mov si, 0` ; регистр si нужен для адресации по массиву
	`mov dl, 'a'` ; нужен для того чтобы поместить в al нужную букву и написать ее

`update_loop:`
	`cmp si, 26` ;условие остановки если si будет 26 индексом (т.е концов алфавита)
	`je update_end` ;прыжок на енд 
	
	mov bl, [turn_symbol_array+si] ; берем текущий индекс
	cmp bl, 0 ; проверяем на 0
	je update_underscore ; если 0 прыжок на нижнее подчеркивание
	
	mov al, dl ;обновляем al
	int 0x10 ; прерывание на вывод
	jmp update_space ; после делаем пробел

`update_underscore:` 
	`mov al, "_` ; печать 
	`int 0x10`
`update_space:`
	`mov al, ' '` ; печать 
	`int 0x10`
	
	`inc si` ; инкремент si для того чтобы взять следующую букву
	`inc dl` ; аналогично si
	`jmp update_loop` ; прыжок на цикл
	
`update_end:` 
	`ret` ; return в основной цикл ввода
