Для начала создадим функцию загрузки ядра:
`run_kernel:` функция после того как человек нажмет ENTER (а значит нужно добавить jmp после проверки на нажатие enter)
![[Pasted image 20260326230108.png]]
	`mov ax, 0x1000` записываем адрес и загружаем ядро
	`mov es, ax` 
	`xor bx, bx`
	
	`mov ah, 0x02`; прерывание на чтение секторов
	`mov al, 0x01` ; количество секторов
	`mov ch, 0x00` ; номер цилиндра
	`mov cl, 0x01` ; номер сектора
	`mov dh, 0x00` ; номер головки
	`mov dl, 0x01` ; ВТОРОЙ диск
	`int 0x13` ; прерывание
	`jc disk_error` ; иначе прыгаем в функцию disk_error
	
	jmp kernel_loaded ; если все ок прыгаем в kernel_load

`kernel_loaded:`
    `jmp 0x1000:0x0000`

`disk_error:` 
	`mov ah, 0x0E` прерывание на вывод текста
	`mov al, 13` ; enter +space
	`int 0x10`
	`mov al, 10` 
	`int 0x10`
	`mov bx, msg_disk_error`пишет сообщение DISK ERROR OR NOT FOUD 
	`call turn_show_str` используем функцию которую писали ранее для текста 
	`jmp $` ;чтобы не провалились в след. функцию завершаем

	jmp input_symbol_loop



`input_symbol_loop:`
	`mov ah, 0x00` 
	`int 0x16`

	cmp al, 13
	je done
	
	cmp al, 'a'
	jb not_a_symbol
	
	cmp al, 'z'
	ja not_a_symbol
	
	sub al, 'a' 
	
	xor ah,ah
	mov si,ax
	
	mov cl, [turn_symbol_array+si]
	xor cl, 1
	mov [turn_symbol_array +si], cl
	
	mov ah, 0x0E
	mov al, 13 
	int 0x10
	mov al, 10 
	int 0x10

	call update_selected_symbols
	jmp input_symbol_loop
	
`update_selected_symbols:`
	`mov ah, 0x0E`
	`mov si, 0` 
	`mov dl, 'a'`

`update_loop:`
	`cmp si, 26`
	`je update_end`
	
	mov cl, [turn_symbol_array+si]
	cmp cl, 0 
	je update_underscore
	
	mov al, dl
	int 0x10
	jmp update_space

`update_underscore:` 
	`mov al, '_'`
	`int 0x10`
`update_space:`
	`mov al, ' '`
	`int 0x10`
	
	inc si
	inc dl
	jmp update_loop
	
`update_end:` 
	`ret`


`run_kernel:`
	`mov ax, 0x1000`
	`mov es, ax`
	`xor bx, bx`
	`mov ah, 0x02`
	`mov al, 0x01`
	`mov ch, 0x00`
	`mov cl, 0x01`
	`mov dh, 0x00`
	`mov dl, 0x01`
	`int 0x13`
	`jc disk_error`
	
	jmp kernel_loaded 

`kernel_loaded:`
    `jmp 0x1000:0x0000`

`disk_error:`
	`mov ah, 0x0E`
	`mov al, 13` 
	`int 0x10`
	`mov al, 10` 
	`int 0x10`
	`mov bx, msg_disk_error`
	`call turn_show_str`
	`jmp $`

`turn_show_str:`
	`mov ah, 0x0E`
	
`print_str_loop:`
		`mov al, [bx]`
		`test al, al`
		`jz print_str_end`
		`int 0x10`
		`inc bx`
		`jmp print_str_loop`
		
`print_str_end:`
	`ret`

`not_a_symbol:` 
	`mov ah, 0x0E`
	`mov al, 13` 
	`int 0x10`
	`mov al, 10` 
	`int 0x10`
	
	mov bx, msg_not_a_symbol	
	call turn_show_str
	jmp input_symbol_loop

done:
	call run_kernel
	jmp $

turn_symbol_array: times 26 db 0
msg db 'Choose and turns symbols...',0
msg_not_a_symbol db 'DONT SYMBOL, Please choose symbol again',0
alphabet db 'a b c d e f g h i j k l m n o p q r s t u v w x y z', 0
complete_choosing db 'Press ENTER then save your choose and next', 0
msg_disk_error db 'DISK ERROR OR NOT FOUND', 0

inf_loop:
