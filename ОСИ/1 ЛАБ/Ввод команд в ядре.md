static inline unsigned char inb(unsigned short port)
{
    unsigned char data;
    asm volatile ("inb %w1, %b0" : "=a"(data) : "Nd"(port));
    return data;
} прерывание для клавиатуры

int str_check(const char* a, const char* b) // проверка на равенство строк
{
    int i = 0;
    while (a[i] != 0 && b[i] != 0)
    {
        if (a[i] != b[i])
        {
            return 0;
        }
        i++;
    }
    if (a[i] == b[i])
    {
        return 1;
    }
    else
    {
        return 0;
    }
}

void backspace() // функция стирки
{
    volatile char* video = (volatile char*)0xB8000;
    if (position > 0)
    {
        position--;
        video[position * 2] = ' ';
        video[position * 2 + 1] = 0x07;

    }
}
