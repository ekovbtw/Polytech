`use16`
`org 0x7C00`

`start:`
    `mov ax, cs`
    `mov ds, ax`
    `mov ss, ax`
    `mov sp, start`
    `mov ah, 0x0E`
	`mov bx, msg_welcome`
	`call turn_show_str`
    	`call newline`
	`mov bx, alphabet`
	`call turn_show_str`
	`call newline`
	`mov bx, complete_choosing`
	`call turn_show_str`
	`call newline`
	`mov bx, msg` 
	`call turn_show_str`
	`jmp input_symbol_loop`



`input_symbol_loop:`
	`mov ah, 0x00` 
	`int 0x16`
	`cmp al, 13`
	`je done`
	`cmp al, 'a'`
	`jb not_a_symbol`
	`cmp al, 'z'`
	`ja not_a_symbol`
	`sub al, 'a'` 
	`xor ah,ah`
	`mov si,ax`
	`mov cl, [turn_symbol_array+si]`
	`xor cl, 1`
	`mov [turn_symbol_array +si], cl`
	`mov ah, 0x0E`
	`mov al, 13` 
	`int 0x10`
	`mov al, 10` 
	`int 0x10`
	`call update_selected_symbols`
	`jmp input_symbol_loop`
	
`update_selected_symbols:`
	`mov ah, 0x0E`
	`mov si, 0` 
	`mov dl, 'a'`

`update_loop:`
	`cmp si, 26`
	`je update_end`
	`mov cl, [turn_symbol_array+si]`
	`cmp cl, 0` 
	`je update_underscore`
	`mov al, dl`
	`int 0x10`
	`jmp update_space`

`update_underscore:` 
	`mov al, '_'`
	`int 0x10`
`update_space:`
	`mov al, ' '`
	`int 0x10`
	`inc si`
	`inc dl`
	`jmp update_loop`
	
`update_end:` 
	`ret`


`run_kernel:`
	`mov ax, 0x1000`
	`mov es, ax`
	`xor bx, bx`
	`mov ah, 0x02`
	`mov al, 0x2C`
	`mov ch, 0x00`
	`mov cl, 0x01`
	`mov dh, 0x00`
	`mov dl, 0x01`
	`int 0x13`
	`jc disk_error`
	`jmp kernel_loaded` 

`kernel_loaded:`
	`jmp turn_protect_mode`

`disk_error:`
	`call newline`
	`mov bx, msg_disk_error`
	`call turn_show_str`
	`jmp $`

`turn_protect_mode:`
    	`cli`
    	`lgdt [gdt_info]`
	`; A20`
	`in al, 0x92`
	`or al, 2`
	`out 0x92, al`
	`; CR0`
	`mov eax, cr0`
	`or eax, 1`
	`mov cr0, eax`
	`jmp 0x08:protect_mode`
	
`[BITS 32]`	
`protect_mode:`
	`mov ax, 0x10`
	`mov ds, ax`
	`mov es, ax`
	`mov ss, ax`
	`mov esp, 0x90000`
	`mov eax, 0x10000`
	`call eax`

`hang:`
    `hlt`
    `jmp hang`


`gdt:`
    `db 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00`
    `db 0xff, 0xff, 0x00, 0x00, 0x00, 0x9A, 0xCF, 0x00`
    `db 0xff, 0xff, 0x00, 0x00, 0x00, 0x92, 0xCF, 0x00`
`gdt_end:`

`gdt_info:`
    `dw gdt_end - gdt - 1`
    `dd gdt`


`[BITS 16]`
`turn_show_str:`
	`mov ah, 0x0E`
	
`print_str_loop:`
		`mov al, [bx]`
		`test al, al`
		`jz print_str_end`
		`int 0x10`
		`inc bx`
		`jmp print_str_loop`
		
`print_str_end:`
	`ret`

`newline:`
	`mov ah, 0x0E`
	`mov al, 13`
	`int 0x10`
	`mov al, 10`
	`int 0x10`
	`ret`


`not_a_symbol:` 
	`mov ah, 0x0E`
	`mov al, 13` 
	`int 0x10`
	`mov al, 10` 
	`int 0x10`
	`mov bx, msg_not_a_symbol`	
	`call turn_show_str`
	`jmp input_symbol_loop`



`done:`
	`mov si, turn_symbol_array`
	`xor ax, ax`
	`mov es, ax`
	`mov di, 0x9000`
	`mov cx, 26`

`copy_params:`
	`mov al, [si]`
	`mov [es:di], al`
	`inc si`
	`inc di`
	`loop copy_params`
	`call run_kernel`
	`jmp $`
	
`turn_symbol_array: times 26 db 0`
`msg db 'Choose and turns symbols...',0`
`msg_not_a_symbol db 'Wrong key',0`
`alphabet db 'a b c d e f g h i j k l m n o p q r s t u v w x y z',0`
`complete_choosing db 'PRESS ENTER',0`
`msg_disk_error db 'Disk error',0`
`msg_welcome db 'WELCOME',0`

`inf_loop:`
    `jmp inf_loop`
    `; Заполняем сектор нулями до 512 байт минус 2 байта сигнатуры`
    `times 512 - ($ - $$) - 2 db 0`

    `; Сигнатура загрузочного сектора`
    `db 0x55, 0xAA`
