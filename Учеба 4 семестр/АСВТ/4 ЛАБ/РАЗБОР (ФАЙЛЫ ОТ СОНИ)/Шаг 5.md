int main(){
	timer_init();
	ADC_init();
	
	DDRA = 0xDF;
	DDRB = DDRC = 0xFF;
	DDRD = 0x90;
	
	// После DDRx нужно добавить:
	MCUCR = (1<<ISC11)|(1<<ISC10)|(1<<ISC01)|(1<<ISC00); // по нарастающему фронту
	GICR  = (1<<INT1)|(1<<INT0);  // разрешить INT0 и INT1
	
	init_garlands();
	sei();
	
	while(1){
		if (mode == 1){
			static uint8_t adc_started = 0;
			if (adc_started == 0) {
				take_adc();
				adc_started = 1;
			}
			if (adc_ready == 1) {
				adc_ready = 0;
				adc_started = 0;
				if (cur_param == 0){
					pb = adc_value * 7/1023 + 1;
					OCR2 = (pb - 1) *255 / 7;
					OCR1B = (8 - pb) *255 / 7;
					ind[0]=13; 
					ind[1]=11; 
					ind[2]=18; 
					ind[3]=pb; 
				}
				else if (cur_param == 1){
					db = adc_value * 3/1023;
					OCR2 = db*255 / 3;
					OCR1B = (3 - db) *255 / 3;
					ind[0]=16;
					ind[1]=11;
					ind[2]=18;
					ind[3]=db;
				}
				else if (cur_param == 2){
					bb = adc_value * 127/1023;
					OCR2 = bb*255 / 127;
					OCR1B = (127 - bb) *255 / 127;
					ind[0]=15;
					ind[1]=11;
					ind[2]=18;
					ind[3]=bb;
				}
			}
		}
	}
}
