В первой строке у нас количество вершин и номер варианта, далее идет матрица.
Код в котором получаем матрицу из задания.

`#define _CRT_SECURE_NO_WARNINGS`
`#include <stdio.h>`
`#include <stdlib.h>`
`#include <string.h>`
`int kol_vershin = 5; // переменная для хранения количества вершин в графе`
`int matrix_D[5][5] = { 0 }; // матрица для хранения расстояний`

`void print_matrix_file(); // отладочная функция для печати матрицы на экран`

`int main()`
`{`
	`FILE* file = fopen("Floid.in", "r");` 
    `if (file == NULL)`
    `{`
        `printf("No find file\n");`
        `return 1;`
    `}`
	`char buffer[100]; // буфер для чтения строк из файла`
    `char skip[1024]; // пропустили`
    `fgets(skip, sizeof(skip), file);`
	`for (int i = 0; i < kol_vershin; i++) // цикл для чтения данных из файла и заполнения матрицы`
    `{`
        `for(int j = 0; j < kol_vershin; j++)`
        `{`
            `fscanf(file, "%s", buffer); // считываем данные из файла и сохраняем в матрицу`
            `if (strcmp(buffer, "*") == 0)`
            `{`
                `matrix_D[i][j] = 999999; // если символ "*", то сохраняем 999999 в матрице`
            `}`
            `else`
            `{`
                `matrix_D[i][j] = atoi(buffer); // иначе сохраняем число, преобразованное из строки`
            `}`
        `}`
    `}`
	`//print_matrix_file();`
    `fclose(file);`
    `return 0;`
`}`

`void print_matrix_file() // функция для печати матрицы на экран (отладочная функция)`
`{`
    `for (int i = 0; i < kol_vershin; i++)`
    `{`
        `for (int j = 0; j < kol_vershin; j++)`
        `{`
            `printf("%d ", matrix_D[i][j]);`
        `}`
        `printf("\n");`
    `}`
`}`
