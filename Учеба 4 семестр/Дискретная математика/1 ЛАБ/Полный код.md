`int matrix_1[58][58] = { 0 };// тут буду хранить строки`

`int array_turn_index[58] = {0}; // тут хранит номера вершин, которые уже есть в графе 1 - turn/ 0 - noturn`

`int from_row = -1; // из какой вершины (индекс)` 

`int E[57][2] = { 0 }; // Множество ребер`

`int summ = 0; // переменная для подсчета суммы веса графа`

`int kol_vershin = 58; // количество вершин (для отладки)`

`int end_matrix[58][58] = { 0 }; // ИТОГОВАЯ МАТРИЦА` 

`int minimum_global = 0; // глобальная переменная для минимума`

`int skip_zero_for_min(int array_min[58],int index) // пропуск нулей ... не обязательная функция`
`{`
    `while (index < kol_vershin && array_min[index] == 0) index++;`
    `return index;`
`}`

`int find_from_index(int min, int row[58]) // ищем индекс минимального значения (для добавления соответствующей вершины)`
`{`
    `for (int i = 0; i < kol_vershin; i++)`
    `{`
        `if (min == row[i])`
        `{`
            `return i;`
        `}`
    `}`
    `return -1;`
`}`

`int find_index_min_in_row(int row[58], int min)`
`{`
    `for (int i = 0; i < kol_vershin; i++)`
    `{`
        `if (row[i] == min && array_turn_index[i] != 1)`
        `{`
            `return i;`
        `}`
    `}`
    `return -1;`
`}`

`int find_min(int array_min[58]) // нахождение минимального значения из всех минимальных`
`{`
    `int min = 0;`
    `int j = 0;`
    `while (j < kol_vershin && array_min[j] == 0) j++;`
    `if (j == kol_vershin) return 0;`
    `min = array_min[j];`
    `for (int i = 0;i < kol_vershin;i++)`
    `{`
        `if (array_min[i] == 0)`
        `{`
            `continue;`
        `}`
        `if (min > array_min[i])`
        `{`
            `min = array_min[i];`
        `}`
    `}`
    `return min;`
`}`

`int find_min_in_row(int row[58]) // поиск только в одной строке (которая включена) минимального значения` 
`{`
    `int j = 0;`
    `while (j < kol_vershin && (row[j] == 0 || array_turn_index[j] == 1)) j++;`
    `if (j == kol_vershin) return 0;`
    `int min = 0;`
    `min = row[j];` 
    `for (int i = 0; i < kol_vershin; i++)`
    `{`
        `if (row[i] == 0)`
        `{`
            `continue;`
        `}`
        `else`
        `{`
            `if (min > row[i] && array_turn_index[i]!=1) // проверка на повтор вершин`
            `{`
                `min = row[i];`
            `}`
        `}`
    `}`
    `return min;`
`}`

`int find_next_row() // поиск среди включенных вершин минимальное значение с каждой, далее выборка минимального из них и нахождение индекса`
`{`
    `int array_min[58] = {0};`
    `int min = 0;`
    `int index_row = 0;`
    `int index_next_row = 0;`
    `for (int i = 0; i < kol_vershin; i++)`
    `{`
        `if (array_turn_index[i] != 0)`
        `{`
            `min = find_min_in_row(matrix_1[i]);`
            `array_min[i] = min;`
        `}`
    `}`
    `min = find_min(array_min);`
    `minimum_global = min;`
    `summ += min;`
    `index_row = find_from_index(min, array_min);`
    `if (index_row == -1) return -1;`
    `index_next_row = find_index_min_in_row(matrix_1[index_row], min);`
    `from_row = index_row;`
    `//printf("Minimum is = %d\t", min);`
    `return index_next_row;`
`}`

`void save_E(int index, int count) // функция для отслеживания ребер, которые добавляем (count -- количество добавлений)`
`{`
    `int old_index = from_row;`
    `int new_index = index;`
    `if (old_index > new_index)`
    `{`
        `int t = old_index;`
        `old_index = new_index;`
        `new_index = t;`
    `}`
    `E[count][0] = old_index;`
    `E[count][1] = new_index;`
`}`

            `{`
                `fprintf(out, ", ");`

        `}`

        `matrix_1[index][i] = matrix_file[index][i];`
    `}`
`}`

`void add_to_end_matrix(int index) // функция для добавления в итоговую матрицу`
`{`
    `int new_index = index; // новый индекс`
    `int old_index = from_row; // откуда`
    `int weight = minimum_global; // какой вес`
    `end_matrix[old_index][new_index] = weight;`
    `end_matrix[new_index][old_index] = weight;`
`}`
