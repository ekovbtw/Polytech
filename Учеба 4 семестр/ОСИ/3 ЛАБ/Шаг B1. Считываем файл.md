### Пункт 1. Считывание входных данных из файла

Тебе нужно открыть файл и считать из него:`input.txt`

- Первая строка — количество потоков (`thread_cnt`)
- Вторая строка — количество элементов массива (`n`)
- Третья строка — целых чисел через пробел (сам массив)`n`

Массив нужно разместить динамически (), потому что его размер заранее неизвестен.`malloc`

После считывания — закрой файл.

int main()
{
    setlocale(LC_ALL, "Russian");
    FILE* file = fopen("input.txt", "r");

    if (file == NULL)
    {
        printf("Не удалось открыть input.txt\n");
        return 1;
    }

    fscanf(file, "%d", &global_count_threads);
    fscanf(file, "%d", &N);
    global_array = (int*)malloc(N * sizeof(int)); // массив, выделение памяти
    if (global_array == NULL)
    {
        printf("Не удалось выделить память под массив чисел\n");
        fclose(file);
        return 1;
    }

	for (int i = 0; i < N; i++) // сканируем числа для быстрой сортировки
    {
        fscanf(file, "%d", &global_array[i]);
    }

	free(global_array); // освобождение памяти
    fclose(file);
    return 0;
}
