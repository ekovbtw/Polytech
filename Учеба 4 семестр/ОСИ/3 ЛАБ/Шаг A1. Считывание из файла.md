Сначала нужно считать количество потоков, количество чисел из файла input.txt

#include <stdio.h>
#include <stdlib.h>
#include <Windows.h>

int main()
{
    FILE* file = fopen("input.txt", "r");
    if (file == NULL)
    {
        printf("Не удалось открыть input.txt\n");
        return 1;
    }
    int count_threads = 0; // количество потоков
    int count_An = 0; // количество чисел
	fscanf(file, "%d", &count_threads);
    fscanf(file, "%d", &count_An);
	int* Array_An = (int*)malloc(count_An * sizeof(int)); // массив где храняться числа
    for (int i = 0; i < count_An; i++)
    {
        fscanf(file, "%d", &Array_An[i]);
    }
    signed long long S;
    fscanf(file, "%lld", &S);

    free(Array_An);
    fclose(file);
    return 0;
}
