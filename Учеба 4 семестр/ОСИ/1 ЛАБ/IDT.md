**IDT = Interrupt Descriptor Table**  
по-русски: **таблица дескрипторов прерываний**.

Слово “дескриптор” тут можно понимать как **описание**.

То есть IDT — это массив из 256 записей, где каждая запись описывает:

- где находится обработчик
- какой сегмент кода использовать
- какого типа это обработчик
- можно ли его вызывать

Методичка прямо говорит, что в защищённом режиме таблица прерываний называется **IDT**, на её местоположение указывает регистр **IDTR**, а в самой таблице лежат не просто адреса, а **дескрипторы обработчиков**

## Почему именно 256 записей

Потому что номер прерывания — это 1 байт.

1 байт = от 0 до 255.  
Значит, максимум 256 разных номеров.


Одна запись IDT — это **не просто число**.  
Это структура из нескольких полей.

В методичке дана структура с такими полями: младшая часть адреса обработчика, селектор сегмента кода, нулевой байт, флаги, старшая часть адреса обработчика

## 1. Адрес обработчика

У тебя есть функция-обработчик, например обработчик клавиатуры.

У любой функции есть адрес в памяти.

Но в записи IDT этот адрес делят на две части:

- младшие 16 бит
- старшие 16 бит

# 2. Селектор сегмента
Когда процессор должен выполнить какой-то код, ему недостаточно просто адреса. Ему ещё нужно знать: **в каком сегменте кода этот адрес находится**.

Для этого и нужен **селектор сегмента**.

Селектор — это маленькое число, которое указывает на запись в GDT. (Я СИЛЬНО НЕ ПОНЯЛ ЧТО ЭТО)

# 3. Флаги
Поле `flags` говорит процессору:

- запись вообще существует или нет
- какого типа обработчик
- какой у него уровень привилегий

1. **P = Present**  
    это бит “запись существует”
2. **Type = тип шлюза**  
    для обычного аппаратного прерывания клавиатуры нужен **interrupt gate**, обычно 32-битный

Методичка перечисляет типы шлюзов и показывает, что для 32-битного шлюза прерывания используется значение типа `1110`, а для регистрации обработчиков в примере используется `0x80 | IDT_TYPE_INTR`

Обычно тебе нужен флаг вида:

- `0x80` — это бит Present
- плюс тип interrupt gate

Поэтому часто получается что-то вроде:

- `0x8E`


# О Структуре 
Компилятор любит “для удобства” вставлять между полями структуры пустые байты для выравнивания.  
Но процессор ждёт формат **строго по байтам**, без самодеятельности.

Поэтому структура IDT-записи должна быть **packed**, чтобы поля шли подряд именно так, как требует процессор. Это и показано в методичке через `__attribute__((packed))` или `#pragma pack(push,1)`

__asm__("jmp kmain");
#define IDT_TYPE_INTR (0x0E)
#define IDT_TYPE_TRAP (0x0F)
// Селектор секции кода, установленный загрузчиком ОС
#define GDT_CS (0x8)
// Структура описывает данные об обработчике прерывания
struct idt_entry
{
	unsigned short base_lo; // Младшие биты адреса обработчика
	unsigned short segm_sel; // Селектор сегмента кода
	unsigned char always0; // Этот байт всегда 0
	unsigned char flags; // Флаги тип. Флаги: P, DPL, Типы - это константы - IDT_TYPE...
	unsigned short base_hi; // Старшие биты адреса обработчика
} __attribute__((packed)); // Выравнивание запрещено
// Структура, адрес которой передается как аргумент команды lidt
struct idt_ptr
{
	unsigned short limit;
	unsigned int base;
} __attribute__((packed)); // Выравнивание запрещено

struct idt_entry g_idt[256]; // Реальная таблица IDT
struct idt_ptr g_idtp; // Описатель таблицы для команды lidt

void default_intr_handler()
{
	asm("pusha");
	// ... (реализация обработки)
	asm("popa; iret");
}
typedef void (*intr_handler)();
void intr_reg_handler(int num, unsigned short segm_sel, unsigned short
flags, intr_handler hndlr)
{
	unsigned int hndlr_addr = (unsigned int) hndlr;
	g_idt[num].base_lo = (unsigned short) (hndlr_addr & 0xFFFF);
	g_idt[num].segm_sel = segm_sel;
	g_idt[num].always0 = 0;
	g_idt[num].flags = flags;
	g_idt[num].base_hi = (unsigned short) (hndlr_addr >> 16);
}
// Функция инициализации системы прерываний: заполнение массива с адресами обработчиков
void intr_init()
{
	int i;
	int idt_count = sizeof(g_idt) / sizeof(g_idt[0]);
	for(i = 0; i < idt_count; i++)
	intr_reg_handler(i, GDT_CS, 0x80 | IDT_TYPE_INTR,
	default_intr_handler); // segm_sel=0x8, P=1, DPL=0, Type=Intr
}
void intr_start()
{
	int idt_count = sizeof(g_idt) / sizeof(g_idt[0]);
	g_idtp.base = (unsigned int) (&g_idt[0]);
	g_idtp.limit = (sizeof (struct idt_entry) * idt_count) - 1;
	asm("lidt %0" : : "m" (g_idtp) );
}
void intr_enable()
{	
	asm("sti");
}
void intr_disable()
{
	asm("cli");
}



extern "C" void kmain()
{
    intr_disable();
    intr_init();
    intr_start();
    volatile char* video = (volatile char*)0xB8000;

    const char* msg = "HELLO FROM KERNEL";
    
    
    for (int i = 0; msg[i] != 0; i++)
    {
        video[i * 2] = msg[i];
        video[i * 2 + 1] = 0x07;
    }

    while (1)
    {
        __asm__ __volatile__("hlt");
    }
   
}
