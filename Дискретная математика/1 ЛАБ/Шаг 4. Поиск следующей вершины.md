Для начала нужно понять, что в матрице у нас (в общем случае) будет не только одна строка, т.е. нужно проходиться по всем строкам и искать минимальное (не равное нулю)
Для этого напишем полный-примерный код:
`int matrix_1[58][58] = { 0 };// тут буду хранить строки`

`int array_turn_index[58] = {0}; // тут хранит номера вершин, которые уже есть в графе 1 - turn/ 0 - noturn`

`int from_row = -1; // индекс предыдущей вершины (индекс)` 

`int skip_zero_for_min(int array_min[58],int index) // пропуск нулей ... не обязательная функция`

`int find_index_via_min(int min, int row[58]) // ищем индекс минимального значения (для добавления соответствующей вершины)`
`{`
    `for (int i = 0; i < 58; i++)`
    `{`
        `if (min == row[i])`
        `{`
            `return i;`
        `}`
    `}`
    `return -1;`
`}`

`int find_index_min_in_row(int row[58], int min)`
`{`
    `for (int i = 0; i < 58; i++)`
    `{`
        `if (row[i] == min && array_turn_index[i] != 1)`
        `{`
            `return i;`
        `}`
    `}`
    `return -1;`
`}`

`int find_min(int array_min[58]) // нахождение минимального значения из всех минимальных`
`{`
    `int min = 0; // итоговое минимальное` 
    `int j = 0; // скипаем нули`
    `while (j < 58 && array_min[j] == 0) j++;`
    `if (j == 58) return 0;`
    `min = array_min[j]; // делаем минимальным первое значение`
    `for (int i = 0;i < 58;i++)`
    `{`
        `if (array_min[i] == 0) // скип нулей`
        `{`
            `continue;`
        `}`
        `if (min > array_min[i])// присваиваем минимальное значение `
        `{`
            `min = array_min[i];`
        `}`
    `}`
    `return min;`
`}`

`int find_min_in_row(int row[58]) // поиск только в одной строке (которая включена) минимального значения` 
`{`
    `int j = 0;`
    `while (j < 58 && (row[j] == 0 || array_turn_index[j] == 1)) j++;`
    `if (j == 58) return 0; // проходимся до того момента пока не найдем число не равное нулю и запоминаем индекс` 
    `int min = 0; // итоговое минимальное (в строке)`
    `min = row[j]; // присвоим минимальному первое попавшееся значение` 
    `for (int i = 0; i < 58; i++)`
    `{`
        `if (row[i] == 0) // если нашли число равное нулю скипаем`
        `{`
            `continue;`
        `}`
        `else`
        `{`
            `if (min > row[i] && array_turn_index[i]!=1) // проверка на повтор вершин`
            `{`
                `min = row[i];`
            `}`
        `}`
    `}`
    `return min;`
`}`

`int find_next_row() // поиск среди включенных вершин минимальное значение с каждой, далее выборка минимального из них и нахождение индекса`

    `int array_min[58] = {0};// массив где будут храниться минимальные значения собранные из всех строк`

    `int index_next_row = 0; // итоговый индекс следующей вершины`
    `for (int i = 0; i < 58; i++) // проходимся по нашему глобальному массиву(включения вершин, см. выше)`
    `{`
        `if (array_turn_index[i] != 0) // если вершина включена, то рассматриваем веса из нее`
        `{`
            `min = find_min_in_row(matrix_1[i]) // смотрим на функцию выше;`
            `array_min[i] = min; // записываем минимальное из строки в массив`
        `}`
    `}`
    `min = find_min(array_min); // находим минимальное среди минимальных см выше`
    `index_row = find_from_index(min, array_min); // запоминаем индекс предыдущей вершины см выше` 
    `index_next_row = find_index_min_in_row(matrix_1[index_row], min); // запоминаем индекс куда идем см выше`
    `from_row = index_row;`
    `if (index_row == -1) return -1;`
    `return index_next_row;` 
`}`
